def prepare_data(file):
    df=pd.read_table(file,delimiter='\t',header=None)
    X=df.iloc[:,:2].values
    Y=df.iloc[:,2:].values
    scaler = StandardScaler()
    x=scaler.fit_transform(X)
    X_train, X_test, y_train, y_test = train_test_split(x,Y,
                                                    test_size=0.30)
    return X_train, X_test, y_train, y_test

def train_models(X_train,y_train,n_neigh):
    bar = progressbar.ProgressBar(maxval=9, \
                    widgets=[progressbar.Bar('=', '[', ']'), ' ', progressbar.Percentage()])
    
    bar.start()
    bar.update(1)
    #KNN
    knn=KNeighborsClassifier(n_neighbors=4)
    knn.fit(X_train,y_train)
    #LDA
    bar.update(2)
    lda=LDA(n_components=2)
    lda.fit(X_train,y_train)
    #random forest
    bar.update(3)
    rf=RandomForestClassifier(n_estimators=10,criterion='entropy')
    rf.fit(X_train,y_train)
    #decision Tree
    bar.update(4)
    dt=DecisionTreeClassifier(criterion='entropy')
    dt.fit(X_train,y_train)
    #linear SVM
    bar.update(5)
    l_svm=SVC(kernel='linear')
    l_svm.fit(X_train,y_train)
    #gausian SVM
    bar.update(6)
    g_svm=SVC(kernel='rbf')
    g_svm.fit(X_train,y_train)
    #lg
    bar.update(7)
    lg=LogisticRegression()
    lg.fit(X_train,y_train)
    #nf
    bar.update(8)
    nf=GaussianNB()
    nf.fit(X_train,y_train)
    #QDA
    bar.update(9)
    qda=QuadraticDiscriminantAnalysis()
    qda.fit(X_train,y_train)
    bar.finish()
    
    return [knn,lda,rf,dt,l_svm,g_svm,lg,nf,qda],  ['knn', 'LDA', 'rendom forest', 'decision tree', 'linear SVM' ,'Guasian SVM' ,'logistic regression', 'naive bayes' ,'QDA']

# Visualising the aggregation data set result's
def plot_model(model,name,X,Y):
    from matplotlib.colors import ListedColormap
    X_set, y_set = X,Y
    X1, X2 = np.meshgrid(np.arange(start = X_set[:, 0].min() - 1, stop = X_set[:, 0].max() + 1, step = 0.01),
                     np.arange(start = X_set[:, 1].min() - 1, stop = X_set[:, 1].max() + 1, step = 0.01))
    plt.contourf(X1, X2, model.predict(np.array([X1.ravel(), X2.ravel()]).T).reshape(X1.shape),
                 alpha = 0.50, cmap = ListedColormap(colors))
    plt.xlim(X1.min(), X1.max())
    plt.ylim(X2.min(), X2.max())

    plt.scatter(X_set[:, 0], X_set[:, 1],c=y_set.ravel(),
                    cmap = ListedColormap(colors))
    plt.title(name)
    plt.xlabel('x1')
    plt.ylabel('X2')
